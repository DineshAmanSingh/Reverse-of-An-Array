#include <stdio.h>
reverseArray(int arr[],int n)
{
    int i,temp;
    int low=0, high=n-1;
    while(low<high)
    {
        temp=arr[low];
        arr[low]=arr[high];
        arr[high]=temp;
        low++;
        high--;
    }
}
int main() {
   int i,num,arr[50];
   printf("Enter size of array \n");
   scanf("%d",&num);
   printf("Enter element in array\n");
   for(i=0;i<=num-1;i++)
   scanf("%d",&arr[i]);
   reverseArray(arr,num);
   printf("Reverse array is...");
   for(i=0;i<=num-1;i++)                    /*  USE THIS for(i=num-1;i>=0;i--) IF NOT CALL FUNCTION reverseArray */
   printf("\n%d",arr[i]);
   return 0;
}
